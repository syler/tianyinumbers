OpenCV的IPlImage保存为CCITT4压缩的TIFF格式

16 September 2010

有一点要知道：提供给 libtiff 和从 libtiff 返回的缓冲区都是每个字节中包含 8 个像素，而8位灰度图是1字节只包含一个像素 OpenCV读入后都是8位灰度图，需要手动进行转换，即将临近的8个像素放入一个字节中，使用位运算即可。 

void save(IplImage* img,const char* file)
{//from http://1few.com/81.html
	uint32	width = 0, length = 0, bufsize;
	uint32	nchannels = 1;	
	int16	depth = 1;	
	uint32  rowsperstrip = (uint32) -1;
	uint16	photometric = PHOTOMETRIC_MINISWHITE;
	uint16	config = PLANARCONFIG_CONTIG;
	uint16	fillorder = FILLORDER_MSB2LSB; 
	unsigned char *buf = NULL;
	uint16 compression =COMPRESSION_CCITTFAX4;

	unsigned char *buffer = (unsigned char *)img->imageDataOrigin;
	rowsperstrip = img->height;
	width = img->width;
	length = img->height;
	nchannels = img->nChannels;
	depth = img->depth;

	TIFF	*tiff=TIFFOpen(file,"w");
	TIFFSetField(tiff, TIFFTAG_BITSPERSAMPLE, 1);// set the size of the channels
	TIFFSetField(tiff, TIFFTAG_COMPRESSION, compression);
	TIFFSetField(tiff, TIFFTAG_FILLORDER, fillorder);
	TIFFSetField(tiff, TIFFTAG_IMAGELENGTH, length);
	TIFFSetField(tiff, TIFFTAG_IMAGEWIDTH, width);
	TIFFSetField(tiff, TIFFTAG_ORIENTATION, ORIENTATION_TOPLEFT);
	TIFFSetField(tiff, TIFFTAG_PLANARCONFIG, config);
	TIFFSetField(tiff, TIFFTAG_PHOTOMETRIC, photometric);
	TIFFSetField(tiff, TIFFTAG_RESOLUTIONUNIT, RESUNIT_INCH);
	TIFFSetField(tiff, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_UINT);
	TIFFSetField(tiff, TIFFTAG_SAMPLESPERPIXEL, nchannels);// set number of channels per pixel

	rowsperstrip = TIFFDefaultStripSize(tiff, rowsperstrip);
	if (rowsperstrip > length) {
		rowsperstrip = length;
	}

	TIFFSetField(tiff, TIFFTAG_ROWSPERSTRIP, rowsperstrip );

	int fax_total_cols=0;
	if(width%8==0)
	{
		fax_total_cols=width/8;
	}
	else
	{
		fax_total_cols=width/8+1;
	}

	int fax_current_col=0;
	unsigned char *row=(unsigned char *)_TIFFmalloc(fax_total_cols);
	unsigned int res=0;
	for(int y=0;yheight;y++)
	{
		for(int x=0;xwidth;x++)
		{
			uchar* cp= (uchar*)(img->imageData+img->widthStep*y);
			int pixel=1;
			if(cp[x]==255)
			{
				pixel=0;
			}
			res=res|(pixel<<(7-x%8));
			if((x>0 && x%8==7) || x==img->width-1)
			{
				row[fax_current_col]=res;
				fax_current_col++;
				res=0;
			}
		}

		if(TIFFWriteScanline(tiff,row,y,0)<0)
		{
			break;
		}

		fax_current_col=0;
	}


	if(row)
	{
		_TIFFfree(row);
	}

	TIFFClose(tiff);
}
