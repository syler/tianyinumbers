C#中运行时调用泛型的方式

3 October 2014

有时候你需要使用到泛型，但是泛型用到的类型只能运行时决定，可以用到这里讲的办法。

使用场景

在Prism中经常看到这样的语句：

eventAggregator.GetEvent<SomeCompositeEvent>().Publish(aninstance);

这条语句通过Prism内置的EventAggregator向所有的Module发出消息，让其他的Module知道这个状态的更新。 但是如果此时你使用eventaggregator的时机是在Prism ready之前，或需要等待一个先决条件。 那么，你只能先保存下这个task。 等条件成立后，再执行这个函数。

如何运行时使用泛型

使用

MakeGenericMethod

示例代码

下面是一个简单代码，显示如何在运行时调用含有generic type的函数。

using System;
using System.Collections.Generic;
using System.Linq;
using System.Reflection;
using System.Text;
using System.Threading;

namespace ConsoleApplication2
{
    public class EventBase
    {
        protected EventBase()
        {
        }
    }

    public interface IEventAggregator
    {
        TEventType GetEvent<TEventType>() where TEventType : EventBase, new();
    }

    public class myEventAggreagator : IEventAggregator
    {
        public TEventType GetEvent<TEventType>() where TEventType : EventBase, new()
        {
            return new TEventType();
        }
    }


    public class MyClassA : EventBase
    {
        public void Say(string msg)
        {
            Console.Write(msg);
        }
    }

    public class MyClassB : EventBase
    {
        public void Show(string msg, bool isSay)
        {
            if (isSay)
            {
                Console.Write(msg);
            }
        }
    }

    public class Task
    {
        public Type TaskType { get; set; }
        public string MethodName { get; set; }
        public object[] Parameters { get; set; }
    }

    class Program
    {
        private static List<Task> tasks = new List<Task>();

        static void Main(string[] args)
        {
            AddTask(new Task { TaskType = typeof(MyClassA), MethodName = "Say", Parameters = new object[] { "hello" } });
            AddTask(new Task { TaskType = typeof(MyClassB), MethodName = "Show", Parameters = new object[] { " world!", true } });
            Thread.Sleep(200);

            // wait for a long time.
            Console.WriteLine("imagine we wait for a looong time and everything is ready.");

            RunAllTask();

            Console.Read();
        }

        static void AddTask(Task task)
        {
            tasks.Add(task);
        }

        static void RunAllTask()
        {
            foreach (var task in tasks)
            {
                Run(task.TaskType, task.MethodName, task.Parameters);
            }
        }

        static void Run(Type type, string methodname, object[] parameter)
        {
            IEventAggregator eventAggregator = new myEventAggreagator();
			
            object obj = typeof(IEventAggregator).GetMethod("GetEvent").MakeGenericMethod(type).Invoke(eventAggregator, null);
			
            MethodInfo methodinfo = type.GetMethod(methodname);
            if (methodinfo != null)
            {
                methodinfo.Invoke(obj, parameter);
            }
        }
    }
}
